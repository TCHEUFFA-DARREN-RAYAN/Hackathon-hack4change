Run this in MySQL Workbench to create the database and user.

-- Drop the existing user first
DROP USER IF EXISTS 'hackathon'@'localhost';

-- Recreate user and database
CREATE USER 'hackathon'@'localhost' IDENTIFIED BY 'Hackathon123';
CREATE DATABASE IF NOT EXISTS CommonGround;
GRANT ALL PRIVILEGES ON CommonGround.* TO 'hackathon'@'localhost';
FLUSH PRIVILEGES;
